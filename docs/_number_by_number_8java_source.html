92 @Override
93 public boolean onFling(MotionEvent e1, MotionEvent e2, float velocityX, float velocityY) {
94 if (e1 == null || e2 == null) {
95 return false;
96 }
97 float diffX = e2.getX() - e1.getX();
98 float diffY = e2.getY() - e1.getY();
99 if (Math.abs(diffX) > Math.abs(diffY)) {
100 if (Math.abs(diffX) > SWIPE_THRESHOLD && Math.abs(velocityX) > SWIPE_VELOCITY_THRESHOLD) {
101 canSwipe = false;
102 if (diffX > 0) {
103 moveRight(WhatToDo.RIGHT);
104 } else {
105 moveLeft(WhatToDo.LEFT);
106 }
107 }
108 } else {
109 if (Math.abs(diffY) > SWIPE_THRESHOLD && Math.abs(velocityY) > SWIPE_VELOCITY_THRESHOLD) {
110 canSwipe = false;
111 if (diffY > 0) {
112 moveDown(WhatToDo.DOWN);
113 } else {
114 moveUp(WhatToDo.UP);
115 }
116 }
117 }
118 return true;
119 }

285 animator.setDuration(SPEED * yd);
